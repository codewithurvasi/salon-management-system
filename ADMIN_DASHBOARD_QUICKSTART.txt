🚀 QUICK START USAGE:

1. Import in your pages:
   import StatsGrid from '@/components/dashboard/Stats/StatsGrid';
   import DataTable from '@/components/dashboard/Tables/DataTable';
   import { useFilters, useSorting } from '@/hooks/useDashboard';

2. Use hooks for state:
   const { filters, updateFilter } = useFilters();
   const { sortedItems } = useSorting(items);

3. Use helpers for formatting:
   import { formatCurrency, formatDate } from '@/utils/dashboardHelpers';
   formatCurrency(1000)    // ₹1,000.00
   formatDate("2026-03-12") // 12 Mar 2026

4. Import mock data:
   import { dashboardStats, appointments } from '@/data/dashboardData';
